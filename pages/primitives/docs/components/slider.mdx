### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-slider
```

Then import the component in your component file.

```jsx
// Slider.tsx
import * as SliderPrimitive from '@interop-ui/react-slider';
```

### Basic example

Create your styled slider components, then pass those styled components into `Slider` via the `as` prop:

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

const SliderRange = styled(SliderPrimitive.Range, {
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'dodgerblue',
  },
});

render(
  <Slider defaultValue={50}>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb />
  </Slider>
);
```

#### Define step size and markers

Use the `step` prop to define how the slider value updates. Add extra decorative elements as the markers.

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

const SliderRange = styled(SliderPrimitive.Range, {
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'dodgerblue',
  },
});

const Markers = styled('div', {
  display: 'flex',
  justifyContent: 'flex-start',
  height: 4,
  position: 'relative',
});
const Marker = styled('div', {
  width: 1,
  height: '100%',
  backgroundColor: 'dodgerblue',
  position: 'absolute',
});

function SliderWithMarkers(props) {
  const [value, setValue] = React.useState(50);

  return (
    <>
      <Slider min={0} max={100} step={10} value={value} onChange={(newValue) => setValue(newValue)}>
        <SliderTrack>
          <SliderRange />
        </SliderTrack>
        <SliderThumb />
      </Slider>
      <Markers>
        <Marker css={{ left: '0%' }} />
        <Marker css={{ left: '10%' }} />
        <Marker css={{ left: '20%' }} />
        <Marker css={{ left: '30%' }} />
        <Marker css={{ left: '40%' }} />
        <Marker css={{ left: '50%' }} />
        <Marker css={{ left: '60%' }} />
        <Marker css={{ left: '70%' }} />
        <Marker css={{ left: '80%' }} />
        <Marker css={{ left: '90%' }} />
        <Marker css={{ left: '100%' }} />
      </Markers>
    </>
  );
}

render(<SliderWithMarkers />);
```

#### Show tooltip on thumb

Control the Slider and compose extra functionality

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

const SliderRange = styled(SliderPrimitive.Range, {
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'dodgerblue',
  },
});

const Tooltip = styled(TooltipPrimitive, {});

function SliderWithTooltip(props) {
  const [value, setValue] = React.useState(50);

  return (
    <div>
      <Slider value={value} onChange={(newValue) => setValue(newValue)}>
        <SliderTrack>
          <SliderRange />
        </SliderTrack>
        <TooltipPrimitive>
          <TooltipPrimitive.Trigger as={SliderThumb} />
          <TooltipPrimitive.Popper side="top">
            <TooltipPrimitive.Content
              style={{
                backgroundColor: 'dodgerblue',
                padding: '5px',
                fontSize: '12px',
                borderRadius: '3px',
              }}
            >
              {value}
            </TooltipPrimitive.Content>
            <TooltipPrimitive.Arrow
              offset={5}
              width={11}
              height={5}
              style={{
                fill: 'dodgerblue',
              }}
            />
          </TooltipPrimitive.Popper>
        </TooltipPrimitive>
      </Slider>
    </div>
  );
}

render(<SliderWithTooltip />);
```

`SliderRange` is the range part. Must live inside `SliderTrack`.
