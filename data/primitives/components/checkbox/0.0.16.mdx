# Checkbox

<Description>
  A control that allows the user to toggle between checked and not
  checked.
</Description>

```jsx hero
import React from 'react';
import { styled } from '@stitches/react';
import { CheckIcon } from '@radix-ui/react-icons';
import { violet } from '@radix-ui/colors';
import * as CheckboxPrimitive from '@radix-ui/react-checkbox';

const StyledCheckbox = styled(CheckboxPrimitive.Root, {
  appearance: 'none',
  border: 'none',
  padding: 0,
  backgroundColor: violet.violet3,
  color: violet.violet10,
  width: '25px',
  height: '25px',
  borderRadius: '4px',
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',

  '&:hover': { backgroundColor: violet.violet4 },
  '&:focus': {
    outline: 'none',
    boxShadow: `inset 0 0 0 1px ${violet.violet7}, 0 0 0 1px ${violet.violet7}`,
  },
});

// Exports
export const Checkbox = React.forwardRef((props, forwardedRef) => (
  <StyledCheckbox {...props} ref={forwardedRef}>
    <CheckboxPrimitive.Indicator>
      <CheckIcon />
    </CheckboxPrimitive.Indicator>
  </StyledCheckbox>
));

// On your app...
export default () => <Checkbox defaultChecked />;
```

## Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-checkbox
```

## Anatomy

Import the components and piece the parts together.

```jsx
import * as Checkbox from '@radix-ui/react-checkbox';

export default () => (
  <Checkbox.Root>
    <Checkbox.Indicator />
  </Checkbox.Root>
);
```

## Examples

### Indeterminate

```jsx collapsed
import { styled } from 'path-to/stitches.config';
import * as Checkbox from '@radix-ui/react-checkbox';
import { DividerHorizontalIcon } from '@radix-ui/react-icons';

const StyledCheckbox = styled(Checkbox.Root, {
  appearance: 'none',
  backgroundColor: 'transparent',
  border: 'none',
  padding: 0,
  boxShadow: 'inset 0 0 0 1px gainsboro',
  width: 15,
  height: 15,
  borderRadius: 2,
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  marginBottom: 10,

  '&:focus': {
    outline: 'none',
    boxShadow: 'inset 0 0 0 1px dodgerblue, 0 0 0 1px dodgerblue',
  },
});

export default () => {
  const [checked, setChecked] = React.useState('indeterminate');

  return (
    <>
      <StyledCheckbox checked={checked} onCheckedChange={setChecked}>
        <Checkbox.Indicator>
          {checked === 'indeterminate' && <DividerHorizontalIcon />}
          {checked === true && <CheckIcon />}
        </Checkbox.Indicator>
      </StyledCheckbox>

      <button
        type="button"
        onClick={() =>
          setChecked((prevIsChecked) =>
            prevIsChecked === 'indeterminate'
              ? false
              : 'indeterminate'
          )
        }
      >
        Toggle indeterminate
      </button>
    </>
  );
};
```
